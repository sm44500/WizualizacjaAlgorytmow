void counting_sort(int tab[], int sorted_tab[], int n)    
{
    int buckets_amount = 0;
    for (int i = 0; i < n; i++)
        if(tab[i] > buckets_amount)
            buckets_amount = tab[i];
    buckets_amount++;

    int buckets[buckets_amount] = {}; 
    
	for(int j = 0; j < n; j++)
		buckets[tab[j]]++;			    
	
    buckets[0]--;
	for(int i = 1; i < buckets_amount; i++)
		buckets[i] += buckets[i-1];            

	for(int j = n-1; j >= 0; j--)
	{
		sorted_tab[buckets[tab[j]]] = tab[j];          
		buckets[tab[j]]--;		  
	}
}

int main()
{
    int tab[] = {9, 2, 5, 1, 6, 2, 3, 9, 2, 8};
    int n = sizeof(tab) / sizeof(tab[0]);

    int sorted_tab[n] = {};
    counting_sort(tab, sorted_tab, n);

    printf("Sorted tab:\n");
    for (int i = 0; i < n; i++)
        printf("%d ", sorted_tab[i]);

    return 0;
}

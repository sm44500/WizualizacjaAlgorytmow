# USELESS academic/classroom example of a linked list implemented in Python.
  # Don't ever consider using something this crude!  Use the built-in list() type!
  
  class Node:
	  def __init__(self, data):
		  self.item = data
		  self.nref = None
		  self.pref = None
  
  
  class DoublyLinkedList:
	  def __init__(self):
		  self.start_node = None
  
  
  def insert_at_start(self, data):
	  # if list is empty
	  if self.start_node is None:
		  new_node = Node(data)
		  self.start_node = new_node
		  print("node inserted")
		  return
	  new_node = Node(data)
	  new_node.nref = self.start_node
	  self.start_node.pref = new_node
	  self.start_node = new_node
  
  
  def insert_at_end(self, data):
	  # if list is empty
	  if self.start_node is None:
		  new_node = Node(data)
		  self.start_node = new_node
		  return
	  n = self.start_node
	  while n.nref is not None:
		  n = n.nref
	  new_node = Node(data)
	  n.nref = new_node
	  new_node.pref = n
  
  
  def insert_after_item(self, x, data):
	  # if list is empty
	  if self.start_node is None:
		  print("List is empty")
		  return
	  else:
		  n = self.start_node
		  while n is not None:
			  if n.item == x:
				  break
			  n = n.nref
		  if n is None:
			  print("item not in the list")
		  else:
			  new_node = Node(data)
			  new_node.pref = n
			  new_node.nref = n.nref
			  if n.nref is not None:
				  n.nref.prev = new_node
			  n.nref = new_node
  
  
  
  def insert_before_item(self, x, data):
	  # if list is empty
	  if self.start_node is None:
		  print("List is empty")
		  return
	  else:
		  n = self.start_node
		  while n is not None:
			  if n.item == x:
				  break
			  n = n.nref
		  if n is None:
			  print("item not in the list")
		  else:
			  new_node = Node(data)
			  new_node.nref = n
			  new_node.pref = n.pref
			  if n.pref is not None:
				  n.pref.nref = new_node
			  n.pref = new_node
  
  
  def traverse_list(self):
	  # if list is empty
	  if self.start_node is None:
		  print("List has no element")
		  return
	  else:
		  n = self.start_node
		  while n is not None:
			  print(n.item , " ")
			  n = n.nref
  
  
  
  def delete_element_by_value(self, x):
	  # if list is empty
	  if self.start_node is None:
		  print("The list has no element to delete")
		  return 
  
	  # if there is only one node in list
	  if self.start_node.nref is None:
		  if self.start_node.item == x:
			  # if seeked node is root 
			  self.start_node = None
		  else:
			  # if there is no node with a such value
			  print("Item not found")
		  return 
  
	  # if seeked node is root
	  if self.start_node.item == x:
		  self.start_node = self.start_node.nref
		  self.start_node.pref = None
		  return
  
	  n = self.start_node
	  while n.nref is not None:
		  if n.item == x:
			  break
		  n = n.nref
	  if n.nref is not None:
		  n.pref.nref = n.nref
		  n.nref.pref = n.pref
	  else:
		  if n.item == x:
			  # if seeked node is end
			  n.pref.nref = None
		  else:
			  # if there is no node with a such value
			  print("Element not found")
  
  
  Dlist = DoublyLinkedList()
  
  insert_at_end(Dlist, 1)
  insert_at_end(Dlist, 2)
  insert_at_end(Dlist, 3)
  traverse_list(Dlist)
  
  insert_at_start(Dlist, 4)
  insert_at_start(Dlist, 5)
  insert_at_start(Dlist, 6)
  traverse_list(Dlist)
  
  delete_element_by_value(Dlist, 4)
  delete_element_by_value(Dlist, 3)
  delete_element_by_value(Dlist, 10)
  traverse_list(Dlist)

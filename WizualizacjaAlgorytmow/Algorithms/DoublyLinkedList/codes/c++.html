Node* DLinkedList::createNode(int value)
{

	temp->value = value;
	temp->prev = nullptr;
	temp->next = nullptr;
	return temp;
}

Node* DLinkedList::getTail()
{
    Node *temp = root;
	// check is list empty ?
	if (temp == nullptr)
		return nullptr;

	while(temp->next != nullptr)
		temp = temp->next;
	return temp;
}

Node *DLinkedList::findNode(int value) 
{
    Node* temp = root;
	while(temp != nullptr)
	{
		if (temp->value == value)
			return temp;
		temp = temp->next;
	}
	return nullptr;
}

Node* DLinkedList::removeNode(int value) 
{
	Node *temp = findNode(value);

	// if there is no node with a such value
	if (temp == nullptr)
	{
		cout<<"Node with value: "<<value<<" doesn't exist!"<<endl;;
		return root;
	}
	// if there is only one node in list 
    else if (temp == root && temp == end)
    {
        delete root;
        root = nullptr;
        end = nullptr;
        return root;
    }
	// if seeked node is root
	else if (temp == root)
	{
		root = root->next;
		delete root->prev;
		root->prev = nullptr;
	}
	// if seeked node is end
	else if (temp == end)
	{
		temp = temp->prev;
		delete temp->next;
		temp->next = nullptr;
        end = temp;
	}
	else
	{
		(temp->prev)->next = temp->next;
		(temp->next)->prev = temp->prev;
		delete temp;	
	}
	return root;
}

Node* DLinkedList::deleteNodes()
{
	// check is list empty ?
	if (root == nullptr)
		return root;
	
	//delete all elements 
	while(root->next != nullptr)
	{
		root = root->next;
		delete root->prev;
	}
	delete root;
    root = nullptr;
    end = nullptr;
	return root;
}

Node* DLinkedList::addNode(int value)
{
	// if list is empty
    if(root == nullptr)
    {
        root = createNode(value);
        end = root;
        return end;
    } 
	end->next = createNode(value);
	(end->next)->prev = end;
	end = end->next;
	return end;
}

void DLinkedList::printNodes() 
{
	// if list is empty
	if(root == nullptr)
		cout<<"Empty list!";

    Node *temp = root;
	while(temp != nullptr)
	{
		cout<<temp->value<<" ";
		temp = temp->next;
	}
	cout<<endl;
}

findNode(int value) 
    Node* temp = root
	while(temp != NULL)
	
		if (temp->value == value)
			return temp
		temp = temp->next
	
	return NULL

	if (temp == NULL)
		printf("Nie istnieje.")
		return root
	
    else if (temp == root && temp == end)
        delete root
        root = NULL
        end = NULL
        return root
    
	else if (temp == root)
		root = root->next
		delete root->prev
		root->prev = NULL
	
	else if (temp == end)
		temp = temp->prev
		delete temp->next
		temp->next = NULL
        end = temp
	else
		(temp->prev)->next = temp->next
		(temp->next)->prev = temp->prev
		delete temp	
	
	return root

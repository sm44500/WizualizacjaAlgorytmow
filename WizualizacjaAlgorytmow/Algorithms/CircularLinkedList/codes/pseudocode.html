Insert_at_Front(n)
    Node *temp = new Node

    temp->data = n
    temp->next = NULL

    if(tail != NULL)
        temp->next = tail->next
        tail->next = temp
    else
        tail = temp 
        tail->next = tail

Delete_at_Front()
    Node *temp
    
    if(tail==NULL)
        delete temp
        return

    temp=tail->next
    if(temp != tail)
        tail->next=temp->next
        delete temp    
    else
        delete temp
        tail = NULL

void Circular_LinkedList::Insert_at_Front(int n)
{
    Node *temp;
    temp = new Node;
    temp->data = n;
    temp->next = nullptr;

    // check is list empty ?
    if(tail != nullptr)
    {
        temp->next = tail->next;
        tail->next = temp;
    }
    else  
    {
        tail = temp; 
        tail->next = tail;
    }
}

void Circular_LinkedList::Display()
{
    Node *front;
    // check is list empty ?
    if(tail==nullptr)
    {
        cout<<"Empty List: \n";
        return;
    }
    front=tail->next;

    cout<<"\tNodes are:\t";
    while(front!=tail)
    {
        cout<<front->data<<" --> ";
        front=front->next;
    }

    if(front==tail)
        cout<<front->data;
    cout<<endl;
}

void Circular_LinkedList::Delete_at_Front()
{
    Node *temp;
    
    // check is list empty ?
    if(tail==nullptr)
        return;
    
    temp=tail->next;

    // is there only one element?
    if(temp != tail)
    {
        tail->next=temp->next;    
        delete temp;
    }
    else
    {
        delete temp;
        tail = nullptr;
    }
}
